import React from 'react';
import { Link } from 'react-router-dom';
import GlassCard from '../UI/GlassCard';
import GlassButton from '../UI/GlassButton';

const DevVaultPage = ({ portfolios, setPortfolios }) => { // Receive portfolios and setter
  // Simulate fetching user portfolios from state/localStorage
  const userPortfolios = portfolios || [];

  const handleDeletePortfolio = (id) => {
    // Simulate deleting a portfolio
    const updatedPortfolios = userPortfolios.filter(p => p.id !== id);
    setPortfolios(updatedPortfolios);
    localStorage.setItem('morphixPortfolios', JSON.stringify(updatedPortfolios)); // Update local storage
  };

  const handleCreateNew = () => {
    // Navigate back to intake to create a new one
    // Or potentially have a quick "duplicate" button on existing ones
    window.location.href = '/intake'; // Force reload/intake for new instance
  };

  return (
    <div className="p-8">
      <div className="flex justify-between items-center mb-8">
        <h2 className="text-3xl font-bold">Dev-Vault</h2>
        <GlassButton onClick={handleCreateNew} variant="primary">+ New Portfolio</GlassButton>
      </div>

      {userPortfolios.length === 0 ? (
        <div className="text-center py-12">
          <p className="text-gray-400 mb-4">You haven't created any portfolios yet.</p>
          <Link to="/intake">
            <GlassButton variant="primary">Build Your First</GlassButton>
          </Link>
        </div>
      ) : (
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
          {userPortfolios.map((portfolio) => (
            <GlassCard key={portfolio.id} tiltOnHover className="p-6 relative">
              <h3 className="text-xl font-semibold mb-2 truncate">{portfolio.name || 'Untitled Portfolio'}</h3>
              <p className="text-gray-400 text-sm mb-4">Created: {new Date(portfolio.createdAt).toLocaleDateString()}</p>
              <div className="flex justify-between">
                 <Link to={`/builder?id=${portfolio.id}`}>
                    <GlassButton size="sm" variant="secondary">Edit</GlassButton>
                 </Link>
                <button
                  onClick={() => handleDeletePortfolio(portfolio.id)}
                  className="text-red-500 hover:text-red-400 text-sm px-3 py-1"
                >
                  Delete
                </button>
              </div>
              {/* Stats Placeholder */}
              <div className="mt-4 pt-4 border-t border-gray-700/50 text-xs text-gray-500">
                <p>Views: {portfolio.views || 0}</p>
                <p>Last Viewed: {portfolio.lastViewed ? new Date(portfolio.lastViewed).toLocaleString() : 'Never'}</p>
              </div>
            </GlassCard>
          ))}
        </div>
      )}
    </div>
  );
};

export default DevVaultPage;
