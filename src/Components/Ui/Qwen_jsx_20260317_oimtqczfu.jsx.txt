import React, { useRef, useEffect } from 'react';
import { motion } from 'framer-motion';

// A component to render the dynamic parallax starfield or mesh
const ParallaxBackground = () => {
  const containerRef = useRef(null);

  // Function to generate stars
  const generateStars = () => {
    const stars = [];
    const numStars = 150; // Adjust density as needed
    for (let i = 0; i < numStars; i++) {
      stars.push(
        <motion.div
          key={i}
          className="absolute rounded-full bg-white"
          style={{
            top: `${Math.random() * 100}%`,
            left: `${Math.random() * 100}%`,
            width: `${Math.random() * 2 + 1}px`, // Random size 1-3px
            height: `${Math.random() * 2 + 1}px`,
            opacity: Math.random() * 0.5 + 0.1, // Random opacity 0.1 - 0.6
          }}
          animate={{
            x: ['var(--bg-x)', 0], // Animate based on parent's --x var
            y: ['var(--bg-y)', 0], // Animate based on parent's --y var
          }}
          transition={{
            type: "tween", // Use tween for direct pixel movement
            duration: 0.5, // Match the mouse move update rate roughly
            ease: "easeOut",
          }}
        />
      );
    }
    return stars;
  };

  return (
    <div
      ref={containerRef}
      className="fixed inset-0 overflow-hidden pointer-events-none z-0" // Behind content, no interaction
    >
      {generateStars()}
      {/* Potentially add a subtle gradient overlay or mesh here if needed */}
      {/* <div className="absolute inset-0 bg-gradient-to-b from-transparent via-black/10 to-black/30"></div> */}
    </div>
  );
};

export default ParallaxBackground;
