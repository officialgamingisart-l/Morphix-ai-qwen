import React, { useEffect, useRef } from 'react';
import { motion } from 'framer-motion';
import ParallaxBackground from '../UI/ParallaxBackground'; // Assume this component exists
import './GlassLayout.css'; // Specific styles for the layout shell

// This component provides the overall "Command Deck" aesthetic for the entire app
const GlassLayout = ({ children }) => {
  const containerRef = useRef(null);

  // Effect to handle mouse-based parallax for the background
  useEffect(() => {
    const handleMouseMove = (e) => {
      if (!containerRef.current) return;
      const { clientX, clientY } = e;
      const { innerWidth, innerHeight } = window;
      const x = (clientX / innerWidth - 0.5) * 10; // Adjust multiplier for intensity
      const y = (clientY / innerHeight - 0.5) * 10; // Adjust multiplier for intensity

      containerRef.current.style.setProperty('--x', `${x}px`);
      containerRef.current.style.setProperty('--y', `${y}px`);
    };

    window.addEventListener('mousemove', handleMouseMove);
    return () => window.removeEventListener('mousemove', handleMouseMove);
  }, []);

  return (
    <div
      ref={containerRef}
      className="relative w-full min-h-screen overflow-hidden"
      style={{
        backgroundColor: '#030303',
        // Apply parallax shift to background elements
        '--bg-x': 'var(--x)',
        '--bg-y': 'var(--y)',
      }}
    >
      {/* Parallax Background Component */}
      <ParallaxBackground />

      {/* Main Content Area - Acts as the Command Deck */}
      <motion.div
        initial={{ opacity: 0 }}
        animate={{ opacity: 1 }}
        transition={{ duration: 0.5 }}
        className="relative z-10 w-full h-full flex flex-col"
      >
        {/* The actual page content gets rendered here */}
        {children}
      </motion.div>
    </div>
  );
};

export default GlassLayout;

// src/components/Layout/GlassLayout.css
/*
This CSS file defines the core "shell" styles for the GlassLayout.
It ensures the backdrop filters, glass panes, and 3D perspective are applied consistently.
*/

/* Ensure the main container has 3D perspective */
.relative.w-full.min-h-screen.overflow-hidden {
  perspective: 1200px;
}

/* Generic Glass Pane Styles - Can be imported/reused */
.glass-pane {
  backdrop-filter: blur(20px) saturate(180%);
  background: rgba(10, 10, 10, 0.4);
  border: 1px solid rgba(255, 255, 255, 0.08);
  margin: 20px;
  border-radius: 24px;
  box-shadow: 0 8px 32px rgba(0, 0, 0, 0.2);
  overflow: hidden;
}

/* Example class for the main content area */
.main-glass-area {
  composes: glass-pane;

  display: flex;

}
